    <script>
    (() => {
        // State 
        // queue: { userId(str) → { name, from, to, endpoint, payload } }
        const queue = {};
        // DOM refs 
        const batchBar      = document.getElementById('batch-bar');
        const batchCount    = document.getElementById('batch-count');
        const batchApplyBtn = document.getElementById('batch-apply-btn');
        const confirmModal  = document.getElementById('confirm-modal');
        const modalBody     = document.getElementById('modal-body');
        const modalBackBtn  = document.getElementById('modal-back-btn');
        const modalApplyBtn = document.getElementById('modal-apply-btn');
        const navGuardModal = document.getElementById('nav-guard-modal');
        const navGuardCount = document.getElementById('nav-guard-count');
        const navGuardLeave = document.getElementById('nav-guard-leave');
        const navGuardStay  = document.getElementById('nav-guard-stay');

        // Helpers 
        function queueSize() { return Object.keys(queue).length; }

        function updateBatchBar() {
            const n = queueSize();
            batchBar.classList.toggle('batch-bar-visible', n > 0);
            batchCount.textContent = n === 1 ? '1 change pending' : `${n} changes pending`;
        }

        function rowForId(userId) {
            return document.querySelector(`.user-row[data-user-id="${userId}"]`);
        }

        function enqueue(userId, entry) {
            queue[userId] = entry;
            updateBatchBar();
            const row = rowForId(userId);
            if (row) {
                row.classList.add('row-queued');
                // show cancel, disable other action-btns on this row
                row.querySelectorAll('.action-btn').forEach(b => {
                    b.disabled = true;
                    b.classList.add('btn-sm-disabled');
                });
                row.querySelectorAll('.dequeue-btn').forEach(b => b.style.display = '');
                row.querySelectorAll('.role-select').forEach(s => s.disabled = true);
                // update status badge preview
                const badge = row.querySelector('[data-status-display]');
                if (badge) {
                    badge.textContent = entry.to;
                    badge.className = 'status-badge status-queued';
                }
            }
        }

        function dequeue(userId) {
            delete queue[userId];
            updateBatchBar();
            const row = rowForId(userId);
            if (row) {
                row.classList.remove('row-queued');
                row.querySelectorAll('.action-btn').forEach(b => {
                    b.disabled = false;
                    b.classList.remove('btn-sm-disabled');
                });
                row.querySelectorAll('.dequeue-btn').forEach(b => b.style.display = 'none');
                // restore role selects
                row.querySelectorAll('.role-select').forEach(s => {
                    s.disabled = false;
                    // if role select, re-check vs original
                    const original = s.dataset.original;
                    if (s.value !== original) {
                        // still dirty — re-enqueue
                        s.value = original; // reset visually
                    }
                });
                // restore status badge
                const badge = row.querySelector('[data-status-display]');
                if (badge) {
                    const orig = badge.dataset.orig || badge.textContent;
                    badge.textContent = orig;
                    // restore class from original text
                    const cls = orig.toLowerCase() === 'pending' ? 'status-pending'
                              : orig.toLowerCase() === 'denied'  ? 'status-denied'
                              : 'status-pending';
                    badge.className = `status-badge ${cls}`;
                }
            }
        }

        // Store original badge text on page load
        document.querySelectorAll('[data-status-display]').forEach(badge => {
            badge.dataset.orig = badge.textContent.trim();
        });

        // Action buttons 
        document.querySelectorAll('.action-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                const row      = btn.closest('.user-row');
                const userId   = row.dataset.userId;
                const userName = row.dataset.userName;
                const from     = btn.dataset.from;
                const to       = btn.dataset.to;
                const endpoint = btn.dataset.endpoint;
                enqueue(userId, { name: userName, from, to, endpoint, payload: null });
            });
        });

        // Role selects 
        document.querySelectorAll('.role-select').forEach(sel => {
            sel.addEventListener('change', () => {
                const row      = sel.closest('.user-row');
                const userId   = row.dataset.userId;
                const userName = row.dataset.userName;
                const original = sel.dataset.original;
                const newRole  = sel.value;

                if (newRole === original) {
                    // reset to original — dequeue if queued
                    if (queue[userId]) dequeue(userId);
                    return;
                }
                // Capitalise for display
                const from = original.charAt(0).toUpperCase() + original.slice(1);
                const to   = newRole.charAt(0).toUpperCase()  + newRole.slice(1);
                enqueue(userId, {
                    name: userName, from, to,
                    endpoint: sel.dataset.endpoint,
                    payload: { role: newRole }
                });
            });
        });

        // Dequeue buttons 
        document.querySelectorAll('.dequeue-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                const row    = btn.closest('.user-row');
                const userId = row.dataset.userId;
                // also reset role select if present
                const sel = row.querySelector('.role-select');
                if (sel) sel.value = sel.dataset.original;
                dequeue(userId);
            });
        });

        // Review & Apply 
        batchApplyBtn.addEventListener('click', openConfirmModal);

        function openConfirmModal() {
            modalBody.innerHTML = '';
            Object.values(queue).forEach(entry => {
                const tr = document.createElement('tr');
                tr.innerHTML = `<td>${entry.name}</td><td class="modal-change-cell"><span class="modal-from">${entry.from}</span><span class="modal-arrow">&#8594;</span><span class="modal-to">${entry.to}</span></td>`;
                modalBody.appendChild(tr);
            });
            // Reset apply button state
            modalApplyBtn.disabled = false;
            modalApplyBtn.textContent = 'Apply Changes';
            document.getElementById('modal-title').textContent = 'Review Changes';
            confirmModal.style.display = 'flex';
        }

        modalBackBtn.addEventListener('click', () => { confirmModal.style.display = 'none'; });

        modalApplyBtn.addEventListener('click', async () => {
            modalApplyBtn.disabled = true;
            modalApplyBtn.textContent = 'Applying…';

            const entries   = Object.entries(queue); // [ [userId, entry], … ]
            const failed    = [];
            const succeeded = [];

            for (const [userId, entry] of entries) {
                try {
                    const opts = {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: entry.payload ? JSON.stringify(entry.payload) : '{}'
                    };
                    const resp = await fetch(entry.endpoint, opts);
                    const data = await resp.json();
                    if (!resp.ok || data.status !== 'success') {
                        failed.push({ entry, message: data.message || `HTTP ${resp.status}` });
                    } else {
                        succeeded.push(entry);
                    }
                } catch (err) {
                    failed.push({ entry, message: 'Network error' });
                }
            }

            if (failed.length === 0) {
                // All good — clear beforeunload guard and reload
                window.onbeforeunload = null;
                window.location.reload();
            } else {
                // Show partial-failure summary
                document.getElementById('modal-title').textContent =
                    `${succeeded.length} applied, ${failed.length} failed`;
                modalBody.innerHTML = '';
                failed.forEach(({ entry, message }) => {
                    const tr = document.createElement('tr');
                    tr.innerHTML = `<td>${entry.name}</td><td class="modal-change-cell modal-change-error">${entry.from} &#8594; ${entry.to} <span class="modal-err-msg">${message}</span></td>`;
                    modalBody.appendChild(tr);
                });
                modalApplyBtn.textContent = 'Reload';
                modalApplyBtn.disabled = false;
                modalApplyBtn.onclick = () => { window.onbeforeunload = null; window.location.reload(); };
            }
        });

        // Sub-tabs 
        document.querySelectorAll('.subtab').forEach(tab => {
            tab.addEventListener('click', () => {
                if (queueSize() > 0) {
                    showNavGuard(() => switchTab(tab.dataset.tab));
                    return;
                }
                switchTab(tab.dataset.tab);
            });
        });

        function switchTab(tabId) {
            document.querySelectorAll('.subtab').forEach(t => t.classList.toggle('subtab-active', t.dataset.tab === tabId));
            document.querySelectorAll('.tab-panel').forEach(p => p.style.display = p.id === `tab-${tabId}` ? '' : 'none');
        }

        // Nav guard 
        let pendingNavTarget = null;

        function showNavGuard(onLeave) {
            const n = queueSize();
            navGuardCount.textContent = `You have ${n} unsaved change${n !== 1 ? 's' : ''}. If you leave, they will be lost.`;
            navGuardModal.style.display = 'flex';
            navGuardLeave.onclick = () => {
                navGuardModal.style.display = 'none';
                window.onbeforeunload = null;
                onLeave();
            };
            navGuardStay.onclick = () => { navGuardModal.style.display = 'none'; };
        }

        // Intercept in-app nav links
        document.querySelectorAll('.nav-link').forEach(link => {
            link.addEventListener('click', e => {
                if (queueSize() === 0) return;
                e.preventDefault();
                const href = link.getAttribute('href');
                if (!href || href === '#') return;
                showNavGuard(() => { window.location.href = href; });
            });
        });

        // Browser-level guard (back, close, address bar)
        window.addEventListener('beforeunload', e => {
            if (queueSize() === 0) return;
            e.preventDefault();
            e.returnValue = '';
        });

    })();
    </script>
